class Worker:
    def __init__(self,x,y,lam,q,stepsize):
        self.x=x
        self.y=y
        self.lam=lam
        self.q=q
        self.stepsize=stepsize
        self.s=x.shape[0]
        self.d=x.shape[1]
        self.w=numpy.zeros((d,1))
        self.initw=numpy.zeros((d,1))
        
    def set_param(self,w):
        self.w=w
        self.initw = w
        
    def loss(self):
        yx=numpy.multiply(self.y, self.x)
        yxw = numpy.dot(yx,self.w)
        vec1=numpy.exp(-yxw)
        vec2=numpy.log(1+vec1)
        return numpy.sum(vec2)

    def gradient_descent(self):
        yx=numpy.multiply(self.y, self.x)
        yxw = numpy.dot(yx,self.w)
        vec1=numpy.exp(yxw)
        vec2=numpy.divide(yx,1+vec1)
        vec3 = -numpy.mean(vec2, axis=0).reshape(self.d, 1)
        g= vec3+lam*self.w
        return g
    
    def stochastic_objective_gradient(self,xi,yi):
        yx =yi*xi  # 1-by-d matrix
        yxw = float(numpy.dot(yx, self.w)) # scalar

        # calculate objective function Q_i
        loss = numpy.log(1 + numpy.exp(-yxw)) # scalar
        reg = lam / 2 * numpy.sum(self.w * self.w) # scalar
        obj = loss + reg

        # calculate stochastic gradient
        g_loss = -yx.T / (1 + numpy.exp(yxw)) # d-by-1 matrix
        g = g_loss + lam * w # d-by-1 matrix
        return g
    
    def gd(self):
        if self.w is None:
            self.w = numpy.zeros((self.d, 1)) # zero initialization
        for t in range(self.q):
            g = self.gradient_descent()
            self.w -= self.stepsize * g
            
    def sgd(self):
        n, d1 = self.x.shape
        if self.w is None:
            self.w = numpy.zeros((d1, 1)) # zero initialization

        for t in range(self.q):
            # randomly shuffle the samples
            rand_indices = numpy.random.permutation(n)
            x_rand = self.x[rand_indices, :]
            y_rand = self.y[rand_indices, :]

            objval = 0 # accumulate the objective values
            for i in range(n):
                xi = x_rand[i, :] # 1-by-d matrix
                yi = float(y_rand[i, :]) # scalar
                g = self.stochastic_objective_gradient(xi,yi)
                self.w -= self.stepsize * g

            self.stepsize *= 0.9 # decrease step size
    
    def ascend_direction_GD(self):
        self.gd()
        return self.initw-self.w
    
    def ascend_direction_SGD(self):
        self.sgd()
        return self.initw-self.w

class Server:
    def __init__(self,m,n,d,alpha):
        self.m=m
        self.n=n
        self.d=d
        self.w=numpy.zeros((d,1))
        self.p=0
        self.loss=0
        self.obj=0
        self.alpha=alpha

    
    def broadcast(self):
        return self.w

    def aggregate(self,pk,losses):
        self.loss=0
        self.p=0
        for k in range(self.m):
            self.p += pk[i]
            self.loss += losses[k]
        self.p/self.m
    
    def agd(self,beta):
        self.w -= self.alpha * self.p
        self.alpha *= 0.9
        
    def objective(self,lam):
        reg = lam/2 * numpy.sum(self.w*self.w)
        self.obj = self.loss/self.n+reg
        return self.obj

import math

def csw(m,x,y,lam,max_iter,stepsize,alpha):
    n,d=x.shape
    s=math.floor(n/m)
    server=Server(m,n,d,alpha)
    workers=[]
    
    for i in range(m):
        indices = list(range(i*s,(i+1)*s))
        worker = Worker(x[indices, :],y[indices , :],lam,max_iter,stepsize)
        workers.append(worker)
        
    return server, workers
